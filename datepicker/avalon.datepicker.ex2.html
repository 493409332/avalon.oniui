datepicker demo

使日历可以通过下拉框切换月份和年份的显示

说明：配置changeMonthAndYear为true使得用户可以通过下拉选项切换月份和年份的显示

                <!DOCTYPE HTML>
                <html>
                <head>
                    <meta charset='utf-8'/>
                    <title>datepicker demo</title>
                    <script src='avalon.js'></script>
                </head>
                <body>
                    <div class="demoShow" ms-controller="demo">
                        <fieldset>
                            <legend>配置日历的日期名列表</legend>
                            <input ms-widget="datepicker, a, $aOpts"/>
                            <p>说明：配置日期列表名为["Sun", "Mon", "Tue", "Wed", "Thu", "Fri", "Sat"]</p>
                        </fieldset>
                        <fieldset>
                            <legend>使日历组件从周日开始显示</legend>
                            <input ms-widget="datepicker" data-datepicker-start-day="0" />
                            <p>说明：配置startDay为0，使得日历从周日开始显示，当然也可以配置为2-6，默认是1</p>
                        </fieldset>
                        <fieldset>
                            <legend>使日历可以通过下拉框切换月份和年份的显示</legend>
                            <input ms-widget="datepicker" data-datepicker-change-month-and-year="true" />
                            <p>说明：配置changeMonthAndYear为true使得用户可以通过下拉选项切换月份和年份的显示</p>
                        </fieldset>
                        <script> 
                            require("datepicker/avalon.datepicker", function(avalon) {
                                var model = avalon.define("demo", function(vm) {
                                    vm.$aOpts = {
                                        dayNames: ["Sun", "Mon", "Tue", "Wed", "Thu", "Fri", "Sat"]
                                    }
                                })
                                avalon.scan();
                            })
                        </script>
                    </div>
                </body>
                </html>

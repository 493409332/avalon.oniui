            require(["progressbar/avalon.progressbar"], function() {
                avalon.define("test", function(vm) {
                    vm.progressbar = {
                        onComplete: function() {
                            avalon.log(this)
                            avalon.log('this bar is complete')
                        }
                        , onChange: function(v) {
                            if(v == 0) this.onComplete()
                        }
                    }
                    vm.$skipArray = ["progressbar"]
                    var v = 0
                        , timer = setInterval(function() {
                            if(avalon.vmodels.$aa.value <= 0) return clearInterval(timer)
                            avalon.vmodels.$aa.value--
                        }, 500)
                })
                avalon.scan()
            })
        </script>

                        require("dialog/avalon.dialog", function(avalon) {
                            avalon.define("demo", function(vm) {
                                vm.show = function( id ){
                                    avalon.vmodels[id].toggle = true;
                                }
                                vm.$ssOpts = {
                                    title: "层上层的dialog",
                                    onSubmit: function() {
                                        alert("submit success !");
                                        console.log(arguments);
                                        console.log(this);
                                    }
                                };

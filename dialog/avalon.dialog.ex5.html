模拟alertdata-dialog-type="alert"）

说明：通过在dom节点中配置data-dialog-type="alert"或者直接在ms-widget中配置opts选项,type为alert的情况下不会显示“关闭”按钮和“取消”按钮

            <!DOCTYPE HTML>
            <html>
            <head>
                <meta charset='utf-8'/>
                <title>模拟alert</title>
                <script src='avalon.js'></script>
            </head>
            <body>
                <div class="demoShow" ms-controller="demo">
                    <fieldset>  
                        <legend>模拟alert</legend>
                            <button ms-click="show('bb')">打开对话框</button>
                            <div ms-widget="dialog,bb,opts">
                                呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵
                            </div>
                    </fieldset>
                    <script> 
                        require("dialog/avalon.dialog", function(avalon) {
                            avalon.define("demo", function(vm) {
                                vm.opts = {
                                    type: 'alert',
                                    title: '警告框demo'
                                };
                                vm.show = function( id ){
                                    avalon.vmodels[id].toggle = true;
                                }
                            })
                            avalon.scan();
                        })
                    </script>
                </div>
            </body>
            </html>

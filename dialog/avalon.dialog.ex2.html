点击弹窗的确定和取消按钮进行自定义回调操作（data-dialog-submit="okfun(close)" data-dialog-cansel="cancelfun(close)"）

说明：通过submit和cansel的配置定义用户的回调操作，注意close参数传递的close方法，区分于字符串

            <!DOCTYPE HTML>
            <html>
            <head>
                <meta charset='utf-8'/>
                <title>点击弹窗的确定和取消按钮进行自定义回调操作</title>
                <script src='avalon.js'></script>
            </head>
            <body>
                <div class="demoShow" ms-controller="demo">
                    <fieldset>  
                        <legend>可自定义回调操作的dialog</legend>
                            <button ms-click="show('aa')">打开对话框</button>
                            <div ms-widget="dialog,aa" data-dialog-title="xxxx" data-dialog-submit="okfun(close)" data-dialog-cancel="cancelfun(close)">
                                呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵呵
                            </div>
                    </fieldset>
                    <script> 
                        require("dialog/avalon.dialog", function($$) {
                            $$.define("demo", function(vm) {
                                vm.show = function( id ){
                                    $$.$ui( id ).show();
                                }
                                vm.okfun = function( close ){
                                    alert('success')
                                    close()
                                }
                                vm.cancelfun = function( close ){
                                    alert('fail')
                                    close()
                                }
                            })
                            $$.scan();
                        })
                    </script>
                </div>
            </body>
            </html>

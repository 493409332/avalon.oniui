cboxpicker demo 1

最基本的cboxpicker组件（ms-widget="cboxpicker"）

说明：只配置ms-widget="cboxpicker"组件，默认组件会显示“全部”选框，不想让其出现的话，可以配置data-cboxpicket-alltext=""

            <!DOCTYPE HTML>
            <html>
            <head>
                <meta charset='utf-8'/>
                <title>xxxxx</title>
                <script src='avalon.js'></script>
            </head>
            <body>
                <div class="demoShow" ms-controller="demo">
                    <fieldset>  
                        <legend>确认框</legend>
                            <button ms-click="show('aa')">打开对话框</button>
                            <button ms-click="changeDialog('aa')">改变对话框内容</button>
                            <button ms-click="changeTitle('aa')">改变对话框title</button>
                            <button ms-click="changeModel('aa')">改变对话框model</button>
                            <div ms-widget="dialog,aa" data-dialog-title="标题" data-dialog-submit="okfun(close)" data-dialog-cancel="cancelfun(close)">
                                可以动态改变对话框的内容，点击不同的按钮，试试看它的灵活性呗
                            </div>
                    </fieldset>
                    <script> 
                        require("dialog/avalon.dialog", function($$) {
                            $$.define("demo", function(vm) {
                                var dialog = "";
                                vm.tpl = '我是tpl';
                                vm.tpl2 = '我是tpl2';
                                vm.show = function( id ){
                                    $$.$ui( id ).show();
                                }
                                vm.okfun = function( close ){
                                    alert('success')
                                    close()
                                }
                                vm.cancelfun = function( close ){
                                    alert('fail')
                                    close()
                                }

                                /**
                                 * 更改dialog的content
                                 */
                                vm.changeDialog = function(id) {
                                    if(!dialog) {
                                        dialog = $$.$ui(id);
                                    }
                                    dialog.setContent('{{tpl2}}');
                                }
                                /**
                                 * 更改dialogtitle
                                 */
                                vm.changeTitle = function(id) {
                                    if(!dialog) {
                                        dialog = $$.$ui(id);
                                    }
                                    dialog.setTitle('改变了dialog的title,起初的title为"标题"');
                                };

                                vm.m = {
                                    tpl: '我变了',
                                    tpl2: '我的内容又变了',
                                    okfun: function( close ) {
                                        alert('我变了');
                                        close();
                                    },
                                    cancelfun: function( close ) {
                                        alert('cancel变了');
                                        close();
                                    }
                                };

                                vm.changeModel = function(id) {
                                    if(!dialog) {
                                        dialog = $$.$ui(id);
                                    }
                                    dialog.setModel(vm.m);
                                };
                            })
                            $$.scan();
                        })
                    </script>
                </div>
            </body>
            </html>

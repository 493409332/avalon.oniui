自定义dialog的width（data-dialog-width="320"）

说明：dialog组件默认宽度为480，可以根据情况自定义配置dialog的宽度

            <!DOCTYPE HTML>
            <html>
            <head>
                <meta charset='utf-8'/>
                <title>自定义dialog的width</title>
                <script src='avalon.js'></script>
            </head>
            <body>
                <div class="demoShow" ms-controller="demo">
                    <fieldset>
                        <legend>设置dialog的宽度(默认为480)</legend>
                        <button ms-click="show('ff')">打开对话框</button>
                        <div ms-widget="dialog,ff" data-dialog-title="配置dialog宽度" data-dialog-width="300">
                            <p>配置dialog宽度为300,默认480</p>
                            <p>配置dialog宽度为300,默认480</p>
                            <p>配置dialog宽度为300,默认480</p>
                            <p>配置dialog宽度为300,默认480</p>
                        </div>
                    </fieldset> 
                    <script> 
                        require("dialog/avalon.dialog", function($$) {
                            $$.define("demo", function(vm) {
                                vm.show = function( id ){
                                    $$.$ui( id ).show();
                                }
                            })
                            $$.scan();
                        })
                    </script>
                </div>
            </body>
            </html>

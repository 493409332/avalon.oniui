<!DOCTYPE html>
<html>
    <head>
        <title>mmRequest组件</title>
        <meta charset="UTF-8">
        <meta name="viewport" content="width=device-width">
        <script src="avalon.js"></script>
        
        <script>
            require(["mmRequest"], function() {
                var vm = avalon.define("test", function(vm) {
                    vm.loadScript = function() {
                        avalon.getScript("ajax.php?callback=callback")
                    }
                    vm.jsonData = []
                    vm.time = ""
                })
                window.callback = function(json) {
                    vm.time = +new Date()
                    vm.jsonData = json
                }
                avalon.scan()
            })
        </script>
    </head>
    <body ms-controller="test">
        <h1>mmRequest-getScript</h1>
        <div>
            <button 
                    ms-click="loadScript">点我加载点数据</button>
            <p>---------</p>
            <p 
               ms-if="!jsonData.length">还没有加载数据</p>
            <p 
               ms-repeat-item="jsonData">
                {{item.name}} - {{time}}
            </p>
        </div>

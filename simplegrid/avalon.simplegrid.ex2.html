columns的配置项使用

columnsOrder用于控制每列的出现顺序，它默认是根据columns中的每个对象的field出现顺序推算出来的

<!DOCTYPE html>
<html>
    <head>
        <title>simplegrid</title>
        <meta charset="UTF-8">
        <meta name="viewport" content="width=device-width">
        <script src="../avalon.js"></script>
        <script src="../highlight/shCore.js">

        </script>
        <script>
            require(["simplegrid/avalon.simplegrid"], function() {
                avalon.define("test", function(vm) {

                    vm.simplegrid = {
                        columns: [
                            {field: "zzz", text: "列三", className: "col3"},
                            {field: "yyy", text: "列二", className: "col2", align: "center"},
                            {field: "xxx", text: "列一", className: "col1"}
                        ],
                        columnsOrder: ["xxx", "yyy", "zzz"],
                        dataSource: [{
                                xxx: 10, yyy: 1, zzz: 1
                            },
                            {
                                xxx: 20, yyy: 2, zzz: 2
                            },
                            {
                                xxx: 30, yyy: 3, zzz: 3
                            },
                            {
                                xxx: 40, yyy: 4, zzz: 4
                            }

                        ]
                    }
                    vm.$skipArray = ["simplegrid"]
                })
                avalon.scan()
            })
        </script>
    </head>
    <body >
      
    <fielset ms-controller="test">
        <legend>columns的配置项使用</legend>
      
        <style>
            .col1{
                background: orange;
            }
            .col2{
                background: orangered;
            }
            .col3{
                background:violet;
            }
        </style>
        <div ms-widget="simplegrid">

        </div>
    </fielset>
</body>

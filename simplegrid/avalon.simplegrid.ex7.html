演示hover效果

要点

IE67下,需要通过getTemplate修改默认模板,为每一个tr加一个ms-hover绑定,对应的类名自己在样式表添加

                     getTemplate: function(str) {
                            if (document.body.hasAttribue) {
                                return str//IE67不支持div:hover效果
                            } else {
                                return str.replace('ms-repeat-row="_data"', 'ms-repeat-row="_data" ms-hover="hover"')
                            }
                        },

CSS

            .ui-grid-tbody tr.hover td,  .ui-grid-tbody tr.hover td div{
                background: rgb(232,245,253)
            }
            .ui-grid-tbody tr:hover td,  .ui-grid-tbody tr:hover td div{
                background: rgb(232,245,253)
            }

演示十字线效果

要点

需要通过tbodyTemplate修改的行模板,为每一个tr加两个事件绑定,在移出移入操作类名

                        tbodyTemplate: function(str) {
                            return str.replace('class="ui-grid-td"', 'class="ui-grid-td" ms-mouseenter="addCross" ms-mouseleave="removeCross"')
                        },
                        addCross: function() {
                            avalon(this.parentNode).addClass("hcross")
                            var index = this.cellIndex
                            if (!cols) {
                                var table = this
                                while (table.tagName != "TABLE") {
                                    table = table.parentNode
                                }
                                cols = table.getElementsByTagName("col")
                            }
                            avalon(cols[index]).addClass("vcross")
                        },
                        removeCross: function() {
                            avalon(this.parentNode).removeClass("hcross")
                            var index = this.cellIndex
                            if (cols) {
                                avalon(cols[index]).removeClass("vcross")
                            }
                        },

CSS

            .ui-grid-tbody tr.hcross td{
                background:gold;
            }
            .ui-grid-tbody .vcross{
                background:goldenrod;
            }

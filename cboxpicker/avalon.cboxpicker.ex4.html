cboxpicker demo 4

配置cboxpicker-fetch获取用户定义的所有选项值（data-cboxpicker-fetch="get_options"）

            <!DOCTYPE HTML>
            <html>
            <head>
                <meta charset='utf-8'/>
                <title>配置duplex选项</title>
                <script src='avalon.js'></script>
            </head>
            <body>
                <div class="demoShow" ms-controller="demo4">
                    <fieldset>
                        <legend>data-cboxpicker-duplex="val"</legend>
                        <ul ms-widget="cboxpicker" data-cboxpicker-duplex="val" data-cboxpicker-alltext="全部" data-cboxpicker-fetch="get_options">
                        </ul>
                        {{ val }}
                    </fieldset>
                </div>
                <script>
                    require("./cboxpicker/avalon.cboxpicker", function(avalon) {
                        var sss = avalon.define('demo4',function(vm){
                            vm.val = ["w1", "w2"];
                            vm.get_options = function( callback ){
                                setTimeout(function(){
                                    sss.val = ["w1","w2","w3","w4"];
                                    callback([
                                        { text : '文字1' , value : 'w1' } ,
                                        { text : '文字2' , value : 'w2' } ,
                                        { text : '文字3' , value : 'w3' } ,
                                        { text : '文字4' , value : 'w4' }
                                    ]);
                                },1000);
                            }
                        });
                        avalon.scan();
                    });
                </script>
            </body>
            </html>
data-cboxpicker-duplex="val"

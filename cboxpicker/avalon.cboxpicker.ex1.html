cboxpicker demo 1

最基本的cboxpicker组件（ms-widget="cboxpicker"）

说明：只配置ms-widget="cboxpicker"组件，默认组件会显示“全部”选框，不想让其出现的话，可以配置data-cboxpicket-alltext=""

            <!DOCTYPE HTML>
            <html>
            <head>
                <meta charset='utf-8'/>
                <title>基本的cboxpicker组件</title>
                <script src='avalon.js'></script>
            </head>
            <body>
                <div class="demoShow" ms-controller="demo1">
                    <fieldset ms-controller="demo1">
                        <legend>ms-widget="cboxpicker"</legend>
                        <ul ms-widget="cboxpicker">
                            <li><input type="checkbox" value="1">选项一</li>
                            <li><input type="checkbox" value="2">选项二</li>
                            <li><input type="checkbox" value="3">选项三</li>
                        </ul>
                    </fieldset>
                </div>
                <script>
                    require("./cboxpicker/avalon.cboxpicker", function(avalon) {
                        var sss = avalon.define('demo1',function(vm){
                        });
                        avalon.scan();
                    });
                </script>
            </body>
            </html>
ms-widget="cboxpicker"

  • 选项一
  • 选项二
  • 选项三

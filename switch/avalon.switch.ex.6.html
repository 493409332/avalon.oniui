<!DOCTYPE html>
<html>
    <head>
        <title>switch组件</title>
        <meta charset="UTF-8">
        <meta name="viewport" content="width=device-width">
        <style type="text/css">
            
        </style>
        <script src="../avalon.js"></script>
        
        <script>
            require(["switch/avalon.switch", "tooltip/avalon.tooltip"], function() {
                var id = 0;
                var vmodel = avalon.define("test", function(vm) {
                    vm["switch"] = {
                        getTemplate: function(tpl) {
                            id++;
                            return tpl.replace(/^<div/, "<div ms-class='checked:checked' ")
                        }
                        , onChange: function(v) {
                            if(avalon.vmodels.$b2) {
                                avalon.vmodels.$b2.content = "单击" + (v ? "关闭" : "打开")
                            }
                        }
                    }
                    vm.$ttOpt = {
                        contentGetter: function(elem) {
                            var elem = avalon(elem)
                            if(elem.hasClass("ui-switch")) return "单击" + (elem.hasClass("checked") ? "关闭" : "打开")
                        },
                        position: "tc"
                        , delegate: true
                    }
                    vm.$skipArray = ["switch"]
                })
                avalon.scan()
            })
        </script>
    </head>
    <body ms-controller="test">
        <h1>switch + tooltip</h1>
        <div ms-widget="tooltip,$b2,$ttOpt">
            <p>---------type=mini---------</p>
            <div class="this-for-test-3">
                <input rank="1" checked="checked" type="checkbox" ms-widget="switch" data-switch-type="mini">
            </div>
            <p>---------type=small---------</p>
            <div class="this-for-test">
                <input rank="2" checked="checked" type="checkbox" ms-widget="switch,$cc" data-switch-type="small">
                <input type="button" value="Toggle" onclick="avalon.vmodels.$cc.toggle()">
            </div>
            <p>---------type=normal---------</p>
            <div class="this-for-test-4">
                <input rank="3" checked="checked" type="checkbox" ms-widget="switch,$dd" data-switch-type="normal">
                <input type="button" value="disable" onclick="avalon.vmodels.$dd.disable()">
                <input type="button" value="enable" onclick="avalon.vmodels.$dd.enable()">
            </div>
            <p>---------type=large---------</p>
            <div class="this-for-test-2">
                <input rank="4" checked="checked" type="checkbox" ms-widget="switch" data-switch-type="large">
            </div>
            <p>------------------</p>
        </div>
        
    </body>
